private static Map parseDataType(Object value) {
    Map<String, Object> map = new LinkedHashMap<String, Object>();
    if (value instanceof Map) {
        return parseMap((Map) value);
    } else if (value instanceof List) {
        return parseArray(value);
    } else if (value instanceof Boolean) {
        map.put("type", DataType.BOOLEAN.type());
        map.put("format", DataType.BOOLEAN.format());
    } else if (value instanceof Integer) {
        map.put("type", DataType.INTEGER.type());
        map.put("format", DataType.INTEGER.format());
    } else if (value instanceof Double) {
        map.put("type", DataType.DOUBLE.type());
        map.put("format", DataType.DOUBLE.format());
    } else if (value instanceof Long) {
        map.put("type", DataType.LONG.type());
        map.put("format", DataType.LONG.format());
    } else if (value instanceof Float) {
        map.put("type", DataType.FLOAT.type());
        map.put("format", DataType.FLOAT.format());
    } else {
        map.put("type", DataType.STRING.type());
        map.put("format", DataType.STRING.format());
    }
    return map;
}
